83 /// Performs the actual function signature rewriting step.
84 LogicalResult BufferizeFuncOpConverter::matchAndRewrite(
85  mlir::FuncOp funcOp, ArrayRef<Value> operands,
86  ConversionPatternRewriter &rewriter) const {
87  auto funcType = funcOp.getType();
88 
89  // Convert function arguments using the provided TypeConverter.
90  TypeConverter::SignatureConversion conversion(funcType.getNumInputs());
91  for (auto argType : llvm::enumerate(funcType.getInputs())) {
92  SmallVector<Type, 2> decomposedTypes, convertedTypes;
93  converter.tryDecomposeType(argType.value(), decomposedTypes);
94  converter.convertTypes(decomposedTypes, convertedTypes);
95  conversion.addInputs(argType.index(), convertedTypes);
96  }
97 
98  // Convert the result types of the function.
99  SmallVector<Type, 2> newResultTypes;
100  newResultTypes.reserve(funcOp.getNumResults());
101  for (Type resultType : funcType.getResults()) {
102  SmallVector<Type, 2> originTypes;
103  converter.tryDecomposeType(resultType, originTypes);
104  for (auto origin : originTypes) {
105  Type converted = converter.convertType(origin);
106  auto kind = converter.getResultConversionKind(origin, converted);
107  if (kind == BufferizeTypeConverter::AppendToArgumentsList) {
108  conversion.addInputs(converted);
109  } else {
110  assert(kind == BufferizeTypeConverter::KeepAsFunctionResult);
111  newResultTypes.push_back(converted);
112  }
113  }
114  }
115 
116  if (failed(rewriter.convertRegionTypes(&funcOp.getBody(), converter,
117  &conversion)))
118  return failure();
119 
120  // Update the signature of the function.
121  rewriter.updateRootInPlace(funcOp, [&] {
122  funcOp.setType(rewriter.getFunctionType(conversion.getConvertedTypes(),
123  newResultTypes));
124  });
125  return success();
126 }

132 namespace {
133 // This class represents a mapping from a result to a list of values and some
134 // results that have not yet constructed. Instead, the indices of these
135 // results in the operation that will be constructed are known. They will be
136 // replaced with the actual values when they are available. The order of
137 // adding to this mapping is important.
138 class CallOpResultMapping {
139 public:
140  CallOpResultMapping() { order = 0; };
141 
142  /// Add an available value to the mapping.
143  void addMapping(Value value) { toValuesMapping.push_back({order++, value}); }
144 
145  /// Add the index of unavailble result value to the mapping.
146  void addMapping(unsigned index) {
147  toIndicesMapping.push_back({order++, index});
148  }
149 
150  /// This method returns the mapping values list. The unknown result values
151  /// that only their indicies are available are replaced with their values.
152  void getMappingValues(ValueRange valuesToReplaceIndices,
153  SmallVectorImpl<Value> &values) {
154  // Append available values to the list.
155  SmallVector<std::pair<unsigned, Value>, 2> res(toValuesMapping.begin(),
156  toValuesMapping.end());
157  // Replace the indices with the actual values.
158  for (const std::pair<unsigned, unsigned> &entry : toIndicesMapping) {
159  assert(entry.second < valuesToReplaceIndices.size() &&
160  "The value index is out of range.");
161  res.push_back({entry.first, valuesToReplaceIndices[entry.second]});
162  }
163  // Sort the values based on their adding orders.
164  llvm::sort(res, [](const std::pair<unsigned, Value> &v1,
165  const std::pair<unsigned, Value> &v2) {
166  return v1.first < v2.first;
167  });
168  // Fill the values.
169  for (const std::pair<unsigned, Value> &entry : res)
170  values.push_back(entry.second);
171  }
172 
173 private:
174  /// Keeping the inserting order of mapping values.
175  int order;
176 
177  /// Containing the mapping values with their inserting orders.
178  SmallVector<std::pair<unsigned, Value>, 2> toValuesMapping;
179 
180  /// Containing the indices of result values with their inserting orders.
181  SmallVector<std::pair<unsigned, unsigned>, 2> toIndicesMapping;
182 };
183 } // namespace

185 /// Performs the actual rewriting step.
186 LogicalResult BufferizeCallOpConverter::matchAndRewrite(
187  CallOp callOp, ArrayRef<Value> operands,
188  ConversionPatternRewriter &rewriter) const {
189 
190  Location loc = callOp.getLoc();
191  OpBuilder builder(callOp);
192  SmallVector<Value, 2> newOperands;
193 
194  // TODO: if the CallOp references a FuncOp that only has a declaration (e.g.
195  // to an externally defined symbol like an external library calls), only
196  // convert if some special attribute is set.
197  // This will allow more control of interop across ABI boundaries.
198 
199  // Create the operands list of the new `CallOp`. It unpacks the decomposable
200  // values if a decompose callback function has been provided by the user.
201  for (auto operand : operands) {
202  SmallVector<Value, 2> values;
203  converter.tryDecomposeValue(builder, loc, operand.getType(), operand,
204  values);
205  newOperands.append(values.begin(), values.end());
206  }
207 
208  // Create the new result types for the new `CallOp` and a mapping from the old
209  // result to new value(s).
210  SmallVector<Type, 2> newResultTypes;
211  SmallVector<CallOpResultMapping, 4> mappings;
212  mappings.resize(callOp.getNumResults());
213  for (auto result : llvm::enumerate(callOp.getResults())) {
214  SmallVector<Type, 2> originTypes;
215  converter.tryDecomposeType(result.value().getType(), originTypes);
216  auto &resultMapping = mappings[result.index()];
217  for (Type origin : originTypes) {
218  Type converted = converter.convertType(origin);
219  auto kind = converter.getResultConversionKind(origin, converted);
220  if (kind == BufferizeTypeConverter::KeepAsFunctionResult) {
221  newResultTypes.push_back(converted);
222  // The result value is not yet available. Its index is kept and it is
223  // replaced with the actual value of the new `CallOp` later.
224  resultMapping.addMapping(newResultTypes.size() - 1);
225  } else {
226  // kind = BufferizeTypeConverter::AppendToArgumentsList
227  MemRefType memref = converted.dyn_cast<MemRefType>();
228  if (!memref)
229  return callOp.emitError("Cannot allocate for a non-Memref type");
230  Value alloc = rewriter.create<AllocOp>(loc, memref);
231  newOperands.push_back(alloc);
232  resultMapping.addMapping(alloc);
233  }
234  }
235  }
236 
237  CallOp newCallOp = rewriter.create<CallOp>(loc, callOp.getCallee(),
238  newResultTypes, newOperands);
239 
240  // Build a replacing value for each result to replace its uses. If a result
241  // has multiple mapping values, it needs to be packed to a single value.
242  OpBuilder nextBuilder(callOp.getOperation()->getNextNode());
243  SmallVector<Value, 2> replacedValues;
244  replacedValues.reserve(callOp.getNumResults());
245  for (unsigned i = 0, e = callOp.getNumResults(); i < e; ++i) {
246  SmallVector<Value, 2> valuesToPack;
247  mappings[i].getMappingValues(newCallOp.getResults(), valuesToPack);
248  if (valuesToPack.empty()) {
249  // No replacement is required.
250  replacedValues.push_back(nullptr);
251  } else if (valuesToPack.size() == 1) {
252  replacedValues.push_back(valuesToPack.front());
253  } else {
254  // Values need to be packed using callback function. The same callback
255  // that is used for materializeArgumentConversion is used for packing.
256  Value packed = converter.materializeArgumentConversion(
257  nextBuilder, loc, callOp.getType(i), valuesToPack);
258  replacedValues.push_back(packed);
259  }
260  }
261  rewriter.replaceOp(callOp, replacedValues);
262  return success();
263 }
