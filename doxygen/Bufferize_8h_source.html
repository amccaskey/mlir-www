8 //
9 // We use the term "bufferize" to mean conversion from tensor types to
10 // memref types.
11 //
12 // Generally speaking, for each op that operates on tensor types, a conversion
13 // pattern needs to be written. The infrastructure in this file assists in
14 // defining these conversion patterns in a composable way.
15 //
16 // Bufferization conversion patterns should generally use the ordinary
17 // conversion pattern classes (e.g. OpConversionPattern). A TypeConverter
18 // (accessible with getTypeConverter()) available on such patterns is sufficient
19 // for most cases (if needed at all).
20 //
21 // But some patterns require access to the extra functions on
22 // BufferizeTypeConverter that don't exist on the base TypeConverter class. For
23 // those cases, BufferizeConversionPattern and its related classes should be
24 // used, which provide access to a BufferizeTypeConverter directly.
25 //
26 //===----------------------------------------------------------------------===//

42 /// A helper type converter class for using inside Buffer Assignment operation
43 /// conversion patterns. The default constructor keeps all the types intact
44 /// except for the ranked-tensor types which is converted to memref types.

89  /// This method registers ResultConversionKind for the mapping from type 'T'
90  /// to type 'U'.
91  template <typename T, typename U>
92  void setResultConversionKind(ResultConversionKind kind) {
93  assert((kind != AppendToArgumentsList ||
94  llvm::is_one_of<U, MemRefType, UnrankedMemRefType>::value) &&
95  "Only the memref typed values can be set to be appended to the "
96  "function argument list at the moment");
97  resultTypeConversions.emplace_back(
98  [=](Type origin, Type input) -> Optional<ResultConversionKind> {
99  if (origin.template isa<T>() && input.template isa<U>())
100  return kind;
101  return llvm::None;
102  });
103  }

106  using DecomposeValueConversionCallFn = std::function<Optional<LogicalResult>(
107  OpBuilder &, Location, Type, Value, SmallVectorImpl<Value> &)>;
108 
109  using DecomposeTypeConversionCallFn =
110  std::function<Optional<LogicalResult>(Type, SmallVectorImpl<Type> &)>;

112  using ResultConversionKindFn =
113  std::function<Optional<ResultConversionKind>(Type, Type)>;

115  /// Generate a wrapper for the given decompose value conversion callback.
116  template <typename T, typename FnT>
117  DecomposeValueConversionCallFn
118  wrapDecomposeValueConversionCallback(FnT &&callback) {
119  return [callback = std::forward<FnT>(callback)](
120  OpBuilder &builder, Location loc, Type type, Value value,
121  SmallVectorImpl<Value> &newValues) -> Optional<LogicalResult> {
122  if (T derivedType = type.dyn_cast<T>())
123  return callback(builder, loc, derivedType, value, newValues);
124  return llvm::None;
125  };
126  }
127 
128  /// Generate a wrapper for the given decompose type conversion callback.
129  template <typename T, typename FnT>
130  DecomposeTypeConversionCallFn
131  wrapDecomposeTypeConversionCallback(FnT &&callback) {
132  return [callback = std::forward<FnT>(callback)](
133  Type type,
134  SmallVectorImpl<Type> &results) -> Optional<LogicalResult> {
135  T derivedType = type.dyn_cast<T>();
136  if (!derivedType)
137  return llvm::None;
138  return callback(derivedType, results);
139  };
140  }
141 
142  SmallVector<ResultConversionKindFn, 2> resultTypeConversions;
143  SmallVector<DecomposeValueConversionCallFn, 2> decomposeValueConversions;
144  SmallVector<DecomposeTypeConversionCallFn, 2> decomposeTypeConversions;

199 /// Rewrites the `ReturnOp` to conform with the changed function signature.
200 /// Operands that correspond to return values and their types have been set to
201 /// AppendToArgumentsList are dropped. In their place, a corresponding copy
202 /// operation from the operand to the target function argument is inserted.
203 template <typename ReturnOpSourceTy, typename ReturnOpTargetTy,
204  typename CopyOpTy>
205 class BufferizeReturnOpConverter
206  : public BufferizeOpConversionPattern<ReturnOpSourceTy> {
207 public:
208  using BufferizeOpConversionPattern<
209  ReturnOpSourceTy>::BufferizeOpConversionPattern;
210 
211  /// Performs the actual return-op conversion step.
212  LogicalResult
213  matchAndRewrite(ReturnOpSourceTy returnOp, ArrayRef<Value> operands,
214  ConversionPatternRewriter &rewriter) const final {
215  Location loc = returnOp.getLoc();
216 
217  // Split the operands depending on whether they need a copy operation or
218  // they remain as operands of the return operation. If an operand is
219  // decomposable and a decompose callback function has been provided by the
220  // user, it will be unpacked.
221  SmallVector<Value, 2> newOperands, needCopyOperands;
222  OpBuilder builder(returnOp);
223  for (auto operand : llvm::enumerate(operands)) {
224  SmallVector<Value, 2> values;
225  this->converter.tryDecomposeValue(builder, loc, operand.value().getType(),
226  operand.value(), values);
227  Type type = returnOp.getOperand(operand.index()).getType();
228  SmallVector<Type, 2> originTypes;
229  this->converter.tryDecomposeType(type, originTypes);
230  for (auto value : llvm::enumerate(values)) {
231  Type origin = originTypes[value.index()];
232  Type converted = value.value().getType();
233  auto kind = this->converter.getResultConversionKind(origin, converted);
234  if (kind == BufferizeTypeConverter::KeepAsFunctionResult)
235  newOperands.push_back(value.value());
236  else
237  // kind = BufferizeTypeConverter::AppendToArgumentsList
238  needCopyOperands.push_back(value.value());
239  }
240  }
241 
242  // Insert Copy operations instead for the operands that have been removed
243  // from operand list and appended to the function arguments list.
244  Block &entryBlock = returnOp.getParentRegion()->front();
245  unsigned numFuncArgs = entryBlock.getNumArguments();
246  if (needCopyOperands.size() > numFuncArgs)
247  return returnOp.emitError(
248  "The number of operands that need Copy operations is more "
249  "than the number of target function arguments.");
250  unsigned destArgNum = numFuncArgs - needCopyOperands.size();
251  rewriter.setInsertionPoint(returnOp);
252  for (Value operand : needCopyOperands) {
253  rewriter.create<CopyOpTy>(loc, operand,
254  entryBlock.getArgument(destArgNum));
255  ++destArgNum;
256  }
257  rewriter.replaceOpWithNewOp<ReturnOpTargetTy>(returnOp, newOperands);
258  return success();
259  }
260 };

339 /// The base class for all BufferPlacement transformations.
340 class BufferPlacementTransformationBase {
341 public:
342  using ValueSetT = BufferAliasAnalysis::ValueSetT;
343 
344  /// Finds a common dominator for the given value while taking the positions
345  /// of the values in the value set into account. It supports dominator and
346  /// post-dominator analyses via template arguments.
347  template <typename DominatorT>
348  static Block *findCommonDominator(Value value, const ValueSetT &values,
349  const DominatorT &doms) {
350  // Start with the current block the value is defined in.
351  Block *dom = value.getParentBlock();
352  // Iterate over all aliases and their uses to find a safe placement block
353  // according to the given dominator information.
354  for (Value childValue : values) {
355  for (Operation *user : childValue.getUsers()) {
356  // Move upwards in the dominator tree to find an appropriate
357  // dominator block that takes the current use into account.
358  dom = doms.findNearestCommonDominator(dom, user->getBlock());
359  }
360  // Take values without any users into account.
361  dom = doms.findNearestCommonDominator(dom, childValue.getParentBlock());
362  }
363  return dom;
364  }
365 
366  /// Returns true if the given operation represents a loop by testing whether
367  /// it implements the `LoopLikeOpInterface` or the `RegionBranchOpInterface`.
368  /// In the case of a `RegionBranchOpInterface`, it checks all region-based
369  /// control-flow edges for cycles.
370  static bool isLoop(Operation *op);
371 
372  /// Constructs a new operation base using the given root operation.
373  BufferPlacementTransformationBase(Operation *op);
374 
375 protected:
376  /// Alias information that can be updated during the insertion of copies.
377  BufferAliasAnalysis aliases;
378 
379  /// Stores all internally managed allocations.
380  BufferPlacementAllocs allocs;
381 
382  /// The underlying liveness analysis to compute fine grained information
383  /// about alloc and dealloc positions.
384  Liveness liveness;
385 };
