20 class ConvertForOpTypes : public OpConversionPattern<ForOp> {
21 public:
22  using OpConversionPattern::OpConversionPattern;
23  LogicalResult
24  matchAndRewrite(ForOp op, ArrayRef<Value> operands,
25  ConversionPatternRewriter &rewriter) const override {
26  SmallVector<Type, 6> newResultTypes;
27  for (auto type : op.getResultTypes()) {
28  Type newType = typeConverter->convertType(type);
29  if (!newType)
30  return rewriter.notifyMatchFailure(op, "not a 1:1 type conversion");
31  newResultTypes.push_back(newType);
32  }
33 
34  // Clone and replace.
35  ForOp newOp = cast<ForOp>(rewriter.clone(*op.getOperation()));
36  newOp.getOperation()->setOperands(operands);
37  for (auto t : llvm::zip(newOp.getResults(), newResultTypes))
38  std::get<0>(t).setType(std::get<1>(t));
39  auto bodyArgs = newOp.getBody()->getArguments();
40  for (auto t : llvm::zip(llvm::drop_begin(bodyArgs, 1), newResultTypes))
41  std::get<0>(t).setType(std::get<1>(t));
42  rewriter.replaceOp(op, newOp.getResults());
43 
44  return success();
45  }
46 };
47 } // namespace

50 class ConvertIfOpTypes : public OpConversionPattern<IfOp> {
51 public:
52  using OpConversionPattern::OpConversionPattern;
53  LogicalResult
54  matchAndRewrite(IfOp op, ArrayRef<Value> operands,
55  ConversionPatternRewriter &rewriter) const override {
56  // TODO: Generalize this to any type conversion, not just 1:1.
57  //
58  // We need to implement something more sophisticated here that tracks which
59  // types convert to which other types and does the appropriate
60  // materialization logic.
61  // For example, it's possible that one result type converts to 0 types and
62  // another to 2 types, so newResultTypes would at least be the right size to
63  // not crash in the llvm::zip call below, but then we would set the the
64  // wrong type on the SSA values! These edge cases are also why we cannot
65  // safely use the TypeConverter::convertTypes helper here.
66  SmallVector<Type, 6> newResultTypes;
67  for (auto type : op.getResultTypes()) {
68  Type newType = typeConverter->convertType(type);
69  if (!newType)
70  return rewriter.notifyMatchFailure(op, "not a 1:1 type conversion");
71  newResultTypes.push_back(newType);
72  }
73 
74  // TODO: Write this with updateRootInPlace once the conversion infra
75  // supports source materializations on ops updated in place.
76  IfOp newOp = cast<IfOp>(rewriter.clone(*op.getOperation()));
77  newOp.getOperation()->setOperands(operands);
78  for (auto t : llvm::zip(newOp.getResults(), newResultTypes))
79  std::get<0>(t).setType(std::get<1>(t));
80  rewriter.replaceOp(op, newOp.getResults());
81  return success();
82  }
83 };
