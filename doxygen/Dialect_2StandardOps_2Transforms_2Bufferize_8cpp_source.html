24 class BufferizeDynamicTensorFromElementsOp
25  : public OpConversionPattern<DynamicTensorFromElementsOp> {
26 public:
27  using OpConversionPattern::OpConversionPattern;
28 
29  LogicalResult
30  matchAndRewrite(DynamicTensorFromElementsOp op, ArrayRef<Value> operands,
31  ConversionPatternRewriter &rewriter) const final {
32  // Allocate memory.
33  Location loc = op.getLoc();
34  DynamicTensorFromElementsOp::Adaptor transformed(operands);
35  RankedTensorType tensorType = op.getType().cast<RankedTensorType>();
36  MemRefType memrefType =
37  MemRefType::get(tensorType.getShape(), tensorType.getElementType());
38  Value result =
39  rewriter.create<AllocOp>(loc, memrefType, transformed.dynamicExtents());
40 
41  // Collect loop bounds.
42  int64_t rank = tensorType.getRank();
43  Value zero = rewriter.create<ConstantIndexOp>(loc, 0);
44  Value one = rewriter.create<ConstantIndexOp>(loc, 1);
45  SmallVector<Value, 4> lowerBounds(rank, zero);
46  SmallVector<Value, 4> steps(rank, one);
47  SmallVector<Value, 4> upperBounds;
48  int nextDynamicIndex = 0;
49  for (int i = 0; i < rank; i++) {
50  Value upperBound =
51  tensorType.isDynamicDim(i)
52  ? transformed.dynamicExtents()[nextDynamicIndex++]
53  : rewriter.create<ConstantIndexOp>(loc, memrefType.getDimSize(i));
54  upperBounds.push_back(upperBound);
55  }
56 
57  // Generate tensor elements with a parallel loop.
58  rewriter.create<scf::ParallelOp>(
59  loc, lowerBounds, upperBounds, steps,
60  [&](OpBuilder &b, Location loc, ValueRange ivs) {
61  BlockAndValueMapping mapping;
62  mapping.map(op.body().getArguments(), ivs);
63  for (auto &nestedOp : op.getBody()->without_terminator())
64  b.clone(nestedOp, mapping);
65  auto yieldOp = cast<YieldOp>(op.getBody()->getTerminator());
66  b.create<StoreOp>(loc, mapping.lookup(yieldOp.value()), result, ivs);
67  b.create<scf::YieldOp>(loc);
68  });
69 
70  rewriter.replaceOp(op, {result});
71  return success();
72  }
73 };
74 } // namespace

106 class BufferizeTensorFromElementsOp
107  : public OpConversionPattern<TensorFromElementsOp> {
108 public:
109  using OpConversionPattern::OpConversionPattern;
110  LogicalResult
111  matchAndRewrite(TensorFromElementsOp op, ArrayRef<Value> operands,
112  ConversionPatternRewriter &rewriter) const override {
113  int numberOfElements = op.elements().size();
114  auto resultType = MemRefType::get(
115  {numberOfElements}, op.getType().cast<TensorType>().getElementType());
116  Value result = rewriter.create<AllocOp>(op.getLoc(), resultType);
117  for (auto element : llvm::enumerate(op.elements())) {
118  Value index =
119  rewriter.create<ConstantIndexOp>(op.getLoc(), element.index());
120  rewriter.create<StoreOp>(op.getLoc(), element.value(), result, index);
121  }
122  rewriter.replaceOp(op, {result});
123  return success();
124  }
125 };
